                                displayPalindromeString(palindromeOrNot, userString);

                            // palindrome the string
                            function palindromeString(userString){

                                let n = userString.length;
                                for (let i = 0; i < n/2; i++) {
                                    // chars match
                                    
                                    if (userString[i] !== userString[n - 1 - i]) { 
                                        // no match
                                        return false; 
                                    }
                                }
                                return true; // strictly equal
                            }


                            // display the palindrome string in the UI
                            function displayPalindromeString(palindromeOrNot, userString){

                                if(palindromeOrNot == false){
                                    // getting the "msg" id, to display the message
                                    document.getElementById("msg").innerHTML = `String Is Not A Palindrome.`;

                                    // Making the alert "visible"
                                    document.getElementById("alert").classList.remove("invisible");
                                }else{
                                    // getting the "msg" id, to display the message
                                    document.getElementById("msg").innerHTML = `The input string is a palindrome = ${userString}`;

                                    // Making the alert "visible"
                                    document.getElementById("alert").classList.remove("invisible");
                                }

                            }

The code is strucutred in the box and next to it is the description of the code.

Function: getValues

This function is responsible to get the entered string from the UI via DOM manipulation, convert that to lower case and then pass that string to check if it is palindrome and display the output message accordingly.

Function: palindromeString

This function is responsible for checking if the entered userString is a plaindrom or not i.e., if it is the same or not when spelled backwards. The logic is to compare the first element with the last, the second element with the second last and so on.

Function: displayPalindromeString

If the output of "palindromeString" is true i.e., the entered string is palindrome, then the success message is printed otherwise a different message is printed saying the entered string is not a palindrome.
